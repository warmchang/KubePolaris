## 数据结构

```typescript
interface PodListItem {
  key: string;
  name: string;
  namespace: string;
  node: string;
  status: 'Running' | 'Pending' | 'Failed' | 'Succeeded' | 'Unknown';
  restartCount: number;
  age: string;
  cpu: string;
  memory: string;
  ip: string;
  labels: Record<string, string>;
  ownerReferences?: OwnerReference[];
  conditions: PodCondition[];
  containerStatuses: ContainerStatus[];

## 数据结构

```typescript
interface WorkloadDetail {
  // 基本信息
  kind: 'Deployment' | 'StatefulSet' | 'DaemonSet' | 'Job' | 'CronJob';
  metadata: K8sMetadata;
  spec: WorkloadSpec;
  status: WorkloadStatus;
  
  // 关联资源
  pods: Pod[];
  replicaSets?: ReplicaSet[];
  services: Service[];
  ingresses: Ingress[];
  configMaps: ConfigMap[];
  secrets: Secret[];
  
  // 监控数据
  metrics: WorkloadMetrics;
  
  // 事件历史
  events: K8sEvent[];
  
  // 版本历史
  revisions: WorkloadRevision[];
}

interface WorkloadMetrics {
  timestamp: string;
  replicas: {
    desired: number;
    current: number;
    ready: number;
    available: number;
  };
  resources: {
    cpu: {
      usage: number;
      requests: number;
      limits: number;
    };
    memory: {
      usage: number;
      requests: number;
      limits: number;
    };
  };
  network: {
    inboundRate: number;
    outboundRate: number;
  };
  performance: {
    qps: number;
    responseTime: number;
    errorRate: number;
    availability: number;
  };
}

interface WorkloadRevision {
  revision: number;
  timestamp: string;
  description: string;
  image: string;
  replicas: number;
  status: 'current' | 'history';
  changeLog: string[];
}
```
